import time
import smtplib
from email.message import EmailMessage
import hashlib
from urllib.request import urlopen

url = 'https://www.farwide.com/cwd-western-states-update/'
response = urlopen(url).read()
currentHash = hashlib.sha224(response).hexdigest()

while True:

    try:

        response = urlopen(url).read()
        currentHash = hashlib.sha224(response).hexdigest()
        time.sleep(100)
        response = urlopen(url).read()
        newHash = hashlib.sha224(response).hexdigest()

        if newHash == currentHash:
            continue

        else:

            msg = EmailMessage()
            msg.set_content(url)
            msg['From'] = 'raghuramankavi@gmail.com'
            msg['To'] = 'rkavi1719@gmail.com@gmail.com'
            msg['Subject'] = 'New Daily Activity Report'
            fromaddr = 'raghuramankavi@gmail.com'
            toaddrs = ['rkavi1719@gmail.com']
            server = smtplib.SMTP('smtp.gmail.com', 587)
            server.starttls()
            server = smtplib.SMTP_SSL('smtp.gmail.com', 465)
            server.login('raghuramankavi@gmail.com', '22kavi66')
            server.send_message(msg)
            server.quit()
            response = urlopen(url).read()
            currentHash = hashlib.sha224(response).hexdigest()
            time.sleep(100)
            continue

    except Exception as e:
        print("something went wrong",e)
